Example 1 - Square and Sawtooth Functions

%Square Wave generation
t=linspace(0,6*pi,31);
x1=3*square(t,60);
x2=3*square(t,30);

figure(1);
subplot(221);
stem(t*5/pi,x1);
xlabel('Time index n');
ylabel('Amplitude');
title('Plot 1');
xticks(0:5:30);
grid;

subplot(222);
stem(t*5/pi,x2);
xlabel('Time index n');
ylabel('Amplitude');
title('Plot 2');
xticks(0:5:30);
grid;

%Sawtooth Wave generation
t=linspace(0,5*pi,51);
x3=2*sawtooth(t);
x4=2*sawtooth(t,0.5);

subplot(223);
stem(t*10/pi,x3);
xlabel('Time index n');
ylabel('Amplitude');
title('Plot 3');
xticks(0:10:50);
xlim([0 50]);
grid;

subplot(224);
stem(t*10/pi,x4);
xlabel('Time index n');
ylabel('Amplitude');
title('Plot 4');
xticks(0:10:50);
xlim([0 50]);
grid;

Example 2 - Sampling Sinusoidal Waveforms

t1= 0:0.001:(1-0.001);
t2= 0:0.1:(1-0.001);

figure(2);
subplot(211);
plot(t1, cos(2*pi*3*t1));
hold on;
stem(t2, cos(2*pi*3*t2));
xlabel('Time index n');
ylabel('Amplitude');
title('Sampled Signal 1');
yticks(-1:0.2:1);

subplot(212);
plot(t1, cos(2*pi*7*t1));
hold on;
stem(t2, cos(2*pi*7*t2));
xlabel('Time index n');
ylabel('Amplitude');
title('Sampled Signal 2');
yticks(-1:0.2:1);

y1= cos(2*pi*3*t2) %Sampling list of first signal
y2= cos(2*pi*7*t2) %Sampling list of second signal
